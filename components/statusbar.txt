import React,{Component, useState} from 'react';
import { View, StyleSheet,Text,TouchableHighlight, ActivityIndicator,Button,StatusBar } from 'react-native';
  
const App = () => {
  const[hide,sethide]=useState(false);
  const [barstyle,setbarstyle]=useState('default');
  
  const toggle=()=>{
    setbarstyle('dark-content');
    setTimeout(()=>{
      setbarstyle('default')
    },5000);
    }

    
 
  return (
   <View style={styles.container}>
      <StatusBar
      backgroundColor={'yellow'}
      barStyle={barstyle}
      hidden={hide}/>
      <Button title='Togle Status Bar' onPress={()=>sethide(!hide)} />
      <Button title='Update Style' onPress={toggle} />

      </View>
    
  )
  
  }
export default App;
